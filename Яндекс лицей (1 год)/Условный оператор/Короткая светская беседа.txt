print('Как ваше настроение?')
text = input()
if ('хорош' in text and 'плох' not in text) and \
   ('?' not in text and 'не' not in text):
    print('Это хорошо, что все хорошо:).')
elif ('плох' in text and 'хорош' not in text) and \
     ('?' not in text and 'не' not in text):
    print('Ничего, все еще будет.')
elif 'не' in text or '?' in text:
    print('Я вас не совсем понимаю.')
