def prime(number):
    abc = True
    for i in range(number):
        if i != 0 and number % i != 0 and i != 1:
            abc = True
        elif i != 0 and i != 1 and number % i == 0:
            abc = False
            break
    if abc:
        return 'Простое число'
    else:
        return 'Составное число'
